SELECT REFM_TABLE_NAME
	,R.REFERENCE_CODE
	,R.REFERENCE_DESC
	,(
		SELECT NAME
		FROM SYSIBM.SYSCOLUMNS C
		WHERE C.TBNAME = M.REFM_TABLE_NAME
			AND C.TBCREATOR = (
				SELECT CURRENT SCHEMA
				FROM SYSIBM.SYSDUMMY1
				)
			AND NAME LIKE '%CODE' FETCH FIRST ROWS ONLY
		) AS ATTRIBUTE_NAME
	,(
		SELECT NAME
		FROM SYSIBM.SYSCOLUMNS C
		WHERE C.TBNAME = M.REFM_TABLE_NAME
			AND C.TBCREATOR = (
				SELECT CURRENT SCHEMA
				FROM SYSIBM.SYSDUMMY1
				)
			AND NAME LIKE '%DESC' FETCH FIRST ROWS ONLY
		) AS ATTRIBUTE_DESC
FROM REFERENCE_MASTER M
JOIN REFM_VW R ON R.REFERENCE_MASTER_ID = M.REFERENCE_MASTER_ID
WHERE REFM_TABLE_NAME IN (
		'REFM_CASE_TYPE'
		,'REFM_CASE_STATUS'
		,'REFM_CASE_IGV_TYPE'
		,'REFM_CASE_CATEGORY_TYPE'
		,'REFM_ASSISTANCE_TYPE'
		,'REFM_GENDER'
		,'REFM_CASE_PARTY_ROLE'
		,'REFM_VERIFY_STATUS'
		,'REFM_COUNTRY'
		,'REFM_EMAIL_TYPE'
		,'REFM_PHONE_TYPE'
		)
ORDER BY REFM_TABLE_NAME, REFERENCE_DESC
