CREATE TABLE INTERGOVERNMENTAL_CASE_DETAIL (
  INTERGOVERNMENTAL_ID INTEGER(4) PRIMARY KEY,
  CASE_ID BIGINT(8), FOREIGN KEY (CASE_ID) REFERENCES CASE_DETAIL (CASE_ID),
  RESPONDENT_ID BIGINT(8), FOREIGN KEY (RESPONDENT_ID) REFERENCES PERSON (PERSON_ID),
  PETITIONER_ID BIGINT(8),
  OTHER_STATE_FIPS_ID INTEGER(4), FOREIGN KEY (OTHER_STATE_FIPS_ID) REFERENCES FEDERAL_INFO_PROCESS_STND (FEDERAL_INFO_PROCESS_STND_ID),
  DOCKET_ID INTEGER(4), FOREIGN KEY (DOCKET_ID) REFERENCES DOCKET (DOCKET_ID),
  COUNTY_CODE CHARACTER(2), FOREIGN KEY (COUNTY_CODE) REFERENCES COUNTY (COUNTY_CODE),
  CASE_STATUS_CODE CHARACTER(1), FOREIGN KEY (CASE_STATUS_CODE) REFERENCES REFM_CASE_STATUS (CASE_STATUS_CODE),
  CASE_IGV_TYPE_CODE CHARACTER(1), FOREIGN KEY (CASE_IGV_TYPE_CODE) REFERENCES REFM_CASE_IGV_TYPE (CASE_IGV_TYPE_CODE),
  OTHER_STATE_CASE_TYPE_CODE CHARACTER(1), FOREIGN KEY (OTHER_STATE_CASE_TYPE_CODE) REFERENCES REFM_OTHER_CASE_TYPE (OTHER_CASE_TYPE_CODE),
  REASON_CODE CHARACTER(5),
  OTHER_STATE_CASE_IDC VARCHAR(15),
  OTHER_STATE_DOCKET_IDC VARCHAR(25),
  PETITIONER_NAME VARCHAR(66),
  RESPONDENT_NAME VARCHAR(66),
  CONTACT_FIRST_NAME VARCHAR(16),
  CONTACT_LAST_NAME VARCHAR(30),
  CONTACT_MIDDLE_NAME VARCHAR(16),
  CONTACT_SUFFIX_NAME CHARACTER(3),
  CONTACT_PHONE_NUMB VARCHAR(15),
  CONTACT_PHONE_EXTN_NUMB CHARACTER(6),
  CONTACT_EML VARCHAR(100),
  CONTACT_FAX_NUMB VARCHAR(15),
  REFERRAL_DATE DATE(4),
  COMMENT_NOTE VARCHAR(4000),
  CREATE_DATE DATE(4),
  EFFECTIVE_DATE DATE(4),
  END_DATE DATE(4),
  TRANSACTION_EVENT_SEQ_NUMB BIGINT(8), FOREIGN KEY (TRANSACTION_EVENT_SEQ_NUMB) REFERENCES GLOBAL_TRANS_EVENT_SEQUENCE (TRANSACTION_EVENT_SEQ_NUMB),
  UPDATED_BY_ID INTEGER(4), FOREIGN KEY (UPDATED_BY_ID) REFERENCES USER_DETAIL (USER_ID),
  BEGIN_VALIDITY_DTTM TIMESTAMP(13,12),
  END_VALIDITY_DTTM TIMESTAMP(13,12),
  SYS_TRANS_START_DTTM TIMESTAMP(13,12),
)
