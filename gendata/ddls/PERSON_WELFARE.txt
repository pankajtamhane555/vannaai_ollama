CREATE TABLE PERSON_WELFARE (
  WELFARE_ID INTEGER(4) PRIMARY KEY,
  PERSON_ID BIGINT(8), FOREIGN KEY (PERSON_ID) REFERENCES PERSON (PERSON_ID),
  CASE_ID BIGINT(8), FOREIGN KEY (CASE_ID) REFERENCES CASE_DETAIL (CASE_ID),
  MODIFICATION_REASON_CODE CHARACTER(2), FOREIGN KEY (MODIFICATION_REASON_CODE) REFERENCES REFM_MODIFICATION_REASON (MODIFICATION_REASON_CODE),
  WELFARE_TYPE_CODE CHARACTER(1), FOREIGN KEY (WELFARE_TYPE_CODE) REFERENCES REFM_WELFARE_TYPE (WELFARE_TYPE_CODE),
  NOMADS_ICI_NUMB INTEGER(4),
  NOMADS_NCP_SEQ_NUMB INTEGER(4),
  NOMADS_UPI_NUMB BIGINT(8),
  MEDICAID_PERSON_NUMB BIGINT(8),
  DCFS_PERSON_NUMB BIGINT(8),
  EFFECTIVE_DATE DATE(4),
  END_DATE DATE(4),
  TRANSACTION_EVENT_SEQ_NUMB BIGINT(8), FOREIGN KEY (TRANSACTION_EVENT_SEQ_NUMB) REFERENCES GLOBAL_TRANS_EVENT_SEQUENCE (TRANSACTION_EVENT_SEQ_NUMB),
  UPDATED_BY_ID INTEGER(4), FOREIGN KEY (UPDATED_BY_ID) REFERENCES USER_DETAIL (USER_ID),
  BEGIN_VALIDITY_DTTM TIMESTAMP(13,12),
  END_VALIDITY_DTTM TIMESTAMP(13,12),
  SYS_TRANS_START_DTTM TIMESTAMP(13,12),
)
