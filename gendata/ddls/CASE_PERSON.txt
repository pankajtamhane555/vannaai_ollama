CREATE TABLE CASE_PERSON (
  CASE_ID BIGINT(8) PRIMARY KEY, FOREIGN KEY (CASE_ID) REFERENCES CASE_DETAIL (CASE_ID),
  PERSON_ID BIGINT(8) PRIMARY KEY, FOREIGN KEY (PERSON_ID) REFERENCES PERSON (PERSON_ID),
  CASE_PARTY_ROLE_CODE CHARACTER(1), FOREIGN KEY (CASE_PARTY_ROLE_CODE) REFERENCES REFM_CASE_PARTY_ROLE (CASE_PARTY_ROLE_CODE),
  CASE_PARTY_STATUS_CODE CHARACTER(1), FOREIGN KEY (CASE_PARTY_STATUS_CODE) REFERENCES REFM_CASE_PARTY_STATUS (CASE_PARTY_STATUS_CODE),
  CASE_PERSON_REASON_STATUS_CODE CHARACTER(1), FOREIGN KEY (CASE_PERSON_REASON_STATUS_CODE) REFERENCES REFM_CASE_PERSON_REASON_STATUS (CASE_PERSON_REASON_STATUS_CODE),
  REP_RELATIONSHIP_CODE CHARACTER(1), FOREIGN KEY (REP_RELATIONSHIP_CODE) REFERENCES REFM_REP_RELATIONSHIP (REP_RELATIONSHIP_CODE),
  FAMILY_VIOLENCE_CODE CHARACTER(1), FOREIGN KEY (FAMILY_VIOLENCE_CODE) REFERENCES REFM_YES_NO (YES_NO_CODE),
  AUTH_REP_NAME VARCHAR(66),
  EFFECTIVE_DATE DATE(4),
  TRANSACTION_EVENT_SEQ_NUMB BIGINT(8), FOREIGN KEY (TRANSACTION_EVENT_SEQ_NUMB) REFERENCES GLOBAL_TRANS_EVENT_SEQUENCE (TRANSACTION_EVENT_SEQ_NUMB),
  UPDATED_BY_ID INTEGER(4), FOREIGN KEY (UPDATED_BY_ID) REFERENCES USER_DETAIL (USER_ID),
  BEGIN_VALIDITY_DTTM TIMESTAMP(13,12),
  END_VALIDITY_DTTM TIMESTAMP(13,12),
  SYS_TRANS_START_DTTM TIMESTAMP(13,12),
)
