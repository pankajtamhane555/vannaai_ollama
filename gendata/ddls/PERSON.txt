CREATE TABLE PERSON (
  PERSON_ID BIGINT(8) PRIMARY KEY,
  EMANCIPATION_STATE_CODE CHARACTER(2), FOREIGN KEY (EMANCIPATION_STATE_CODE) REFERENCES ADDRESS_STATE (STATE_CODE),
  BIRTH_CITY_ID INTEGER(4), FOREIGN KEY (BIRTH_CITY_ID) REFERENCES CITY (CITY_ID),
  GENDER_CODE CHARACTER(1), FOREIGN KEY (GENDER_CODE) REFERENCES REFM_GENDER (GENDER_CODE),
  RACE_TYPE_CODE CHARACTER(1), FOREIGN KEY (RACE_TYPE_CODE) REFERENCES REFM_RACE_TYPE (RACE_TYPE_CODE),
  HAIR_COLOR_CODE CHARACTER(2), FOREIGN KEY (HAIR_COLOR_CODE) REFERENCES REFM_HAIR_COLOR (HAIR_COLOR_CODE),
  EYE_COLOR_CODE CHARACTER(2), FOREIGN KEY (EYE_COLOR_CODE) REFERENCES REFM_EYE_COLOR (EYE_COLOR_CODE),
  PRIMARY_LANGUAGE_CODE CHARACTER(2), FOREIGN KEY (PRIMARY_LANGUAGE_CODE) REFERENCES REFM_LANGUAGE (LANGUAGE_CODE),
  DEATH_VERIFY_SOURCE_CODE CHARACTER(1), FOREIGN KEY (DEATH_VERIFY_SOURCE_CODE) REFERENCES REFM_DEATH_VERIFY_SOURCE (DEATH_VERIFY_SOURCE_CODE),
  EMANCIPATION_REASON_CODE CHARACTER(1), FOREIGN KEY (EMANCIPATION_REASON_CODE) REFERENCES REFM_EMANCIPATION_REASON (EMANCIPATION_REASON_CODE),
  CONTACT_METHOD_CODE CHARACTER(1), FOREIGN KEY (CONTACT_METHOD_CODE) REFERENCES REFM_CONTACT_METHOD (CONTACT_METHOD_CODE),
  EDUCATION_LEVEL_CODE CHARACTER(2), FOREIGN KEY (EDUCATION_LEVEL_CODE) REFERENCES REFM_EDUCATION_LEVEL (EDUCATION_LEVEL_CODE),
  TRIBAL_AFFILIATION_CODE CHARACTER(3), FOREIGN KEY (TRIBAL_AFFILIATION_CODE) REFERENCES REFM_TRIBAL_AFFILIATION (TRIBAL_AFFILIATION_CODE),
  SUFFIX_NAME_CODE CHARACTER(3), FOREIGN KEY (SUFFIX_NAME_CODE) REFERENCES REFM_SUFFIX_NAME (SUFFIX_NAME_CODE),
  BIRTH_SOURCE_CODE CHARACTER(3), FOREIGN KEY (BIRTH_SOURCE_CODE) REFERENCES REFM_BIRTH_SOURCE (BIRTH_SOURCE_CODE),
  FACIAL_HAIR_CODE CHARACTER(1), FOREIGN KEY (FACIAL_HAIR_CODE) REFERENCES REFM_YES_NO (YES_NO_CODE),
  LAST_NAME VARCHAR(30),
  FIRST_NAME VARCHAR(16),
  MIDDLE_NAME VARCHAR(16),
  HEIGHT_FEET_NUMB INTEGER(4),
  HEIGHT_INCH_NUMB INTEGER(4),
  WEIGHT_NUMB INTEGER(4),
  DEATH_DATE DATE(4),
  IDENTIFICATION_MARK_DESC VARCHAR(4000),
  GRADUATION_DATE DATE(4),
  DEATH_CERTIFICATE_IDC VARCHAR(20),
  NOMADS_PERSON_NUMB BIGINT(8),
  MAIDEN_NAME VARCHAR(30),
  ADDITIONAL_COMMENT_NOTE VARCHAR(4000),
  EMANCIPATION_DATE DATE(4),
  BIRTH_DATE DATE(4),
  VERIFICATION_DATE DATE(4),
  UNKNOWN_DOB_IND BOOLEAN(1),
  TRANSACTION_EVENT_SEQ_NUMB BIGINT(8), FOREIGN KEY (TRANSACTION_EVENT_SEQ_NUMB) REFERENCES GLOBAL_TRANS_EVENT_SEQUENCE (TRANSACTION_EVENT_SEQ_NUMB),
  UPDATED_BY_ID INTEGER(4), FOREIGN KEY (UPDATED_BY_ID) REFERENCES USER_DETAIL (USER_ID),
  BEGIN_VALIDITY_DTTM TIMESTAMP(13,12),
  END_VALIDITY_DTTM TIMESTAMP(13,12),
  SYS_TRANS_START_DTTM TIMESTAMP(13,12),
  BIRTH_PLACE_RESERVATION_IND BOOLEAN(1),
)
