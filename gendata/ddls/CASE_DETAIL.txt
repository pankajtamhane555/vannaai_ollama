CREATE TABLE CASE_DETAIL (
  CASE_ID BIGINT(8) PRIMARY KEY,
  APPLICATION_ID BIGINT(8), FOREIGN KEY (APPLICATION_ID) REFERENCES APPLICANT_DETAIL (APPLICATION_ID),
  COUNTY_CODE CHARACTER(2), FOREIGN KEY (COUNTY_CODE) REFERENCES COUNTY (COUNTY_CODE),
  CASE_USER_ID INTEGER(4), FOREIGN KEY (CASE_USER_ID) REFERENCES USER_DETAIL (USER_ID),
  CASE_TYPE_CODE CHARACTER(1), FOREIGN KEY (CASE_TYPE_CODE) REFERENCES REFM_CASE_TYPE (CASE_TYPE_CODE),
  SOURCE_REFERRAL_CODE CHARACTER(1), FOREIGN KEY (SOURCE_REFERRAL_CODE) REFERENCES REFM_SOURCE_REFERRAL (SOURCE_REFERRAL_CODE),
  COOPERATION_CODE CHARACTER(1), FOREIGN KEY (COOPERATION_CODE) REFERENCES REFM_COOPERATION (COOPERATION_CODE),
  GOOD_CAUSE_DETERMINATION_CODE CHARACTER(1), FOREIGN KEY (GOOD_CAUSE_DETERMINATION_CODE) REFERENCES REFM_GOOD_CAUSE_DETERMINATION (GOOD_CAUSE_DETERMINATION_CODE),
  GOOD_CAUSE_CLAIM_REASON_CODE CHARACTER(1), FOREIGN KEY (GOOD_CAUSE_CLAIM_REASON_CODE) REFERENCES REFM_GOOD_CAUSE (GOOD_CAUSE_CODE),
  CASE_STATUS_CODE CHARACTER(1), FOREIGN KEY (CASE_STATUS_CODE) REFERENCES REFM_CASE_STATUS (CASE_STATUS_CODE),
  CASE_IGV_TYPE_CODE CHARACTER(1), FOREIGN KEY (CASE_IGV_TYPE_CODE) REFERENCES REFM_CASE_IGV_TYPE (CASE_IGV_TYPE_CODE),
  IVD_APPLICANT_CODE CHARACTER(1), FOREIGN KEY (IVD_APPLICANT_CODE) REFERENCES REFM_IVD_APPLICANT (IVD_APPLICANT_CODE),
  CASE_CATEGORY_TYPE_CODE CHARACTER(2), FOREIGN KEY (CASE_CATEGORY_TYPE_CODE) REFERENCES REFM_CASE_CATEGORY_TYPE (CASE_CATEGORY_TYPE_CODE),
  PRIMARY_FUNCTION_CODE CHARACTER(2), FOREIGN KEY (PRIMARY_FUNCTION_CODE) REFERENCES REFM_PRIMARY_FUNCTION (PRIMARY_FUNCTION_CODE),
  ASSISTANCE_TYPE_CODE CHARACTER(2), FOREIGN KEY (ASSISTANCE_TYPE_CODE) REFERENCES REFM_ASSISTANCE_TYPE (ASSISTANCE_TYPE_CODE),
  OFFICE_CODE CHARACTER(2), FOREIGN KEY (OFFICE_CODE) REFERENCES OFFICE (OFFICE_CODE),
  ASSIGNED_OFFICE_CODE CHARACTER(2), FOREIGN KEY (ASSIGNED_OFFICE_CODE) REFERENCES OFFICE (OFFICE_CODE),
  CASE_CLOSURE_REASON_CODE CHARACTER(3), FOREIGN KEY (CASE_CLOSURE_REASON_CODE) REFERENCES REFM_CASE_CLOSURE_REASON (CASE_CLOSURE_REASON_CODE),
  ENFORCE_STATUS_CODE CHARACTER(1), FOREIGN KEY (ENFORCE_STATUS_CODE) REFERENCES REFM_YES_NO (YES_NO_CODE),
  APPLICATION_FEE_TYPE_DESC VARCHAR(70),
  GOOD_CAUSE_CLAIM_DATE DATE(4),
  GOOD_CAUSE_DETERMINATION_DATE DATE(4),
  OPENED_DATE DATE(4),
  COOPERATION_DATE DATE(4),
  CASE_SOURCE_REFERRAL_DATE DATE(4),
  CASE_STATUS_DATE DATE(4),
  APPLICATION_REQUESTED_DATE DATE(4),
  APPLICATION_SENT_DATE DATE(4),
  APPLICATION_RECEIVED_DATE DATE(4),
  APPLICATION_SIGNED_DATE DATE(4),
  NOMADS_CASE_IDC VARCHAR(10),
  TRANSACTION_EVENT_SEQ_NUMB BIGINT(8), FOREIGN KEY (TRANSACTION_EVENT_SEQ_NUMB) REFERENCES GLOBAL_TRANS_EVENT_SEQUENCE (TRANSACTION_EVENT_SEQ_NUMB),
  UPDATED_BY_ID INTEGER(4), FOREIGN KEY (UPDATED_BY_ID) REFERENCES USER_DETAIL (USER_ID),
  BEGIN_VALIDITY_DTTM TIMESTAMP(13,12),
  END_VALIDITY_DTTM TIMESTAMP(13,12),
  SYS_TRANS_START_DTTM TIMESTAMP(13,12),
)
