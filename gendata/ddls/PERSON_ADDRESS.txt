CREATE TABLE PERSON_ADDRESS (
  PERSON_ADDRESS_ID INTEGER(4) PRIMARY KEY,
  PERSON_ID BIGINT(8), FOREIGN KEY (PERSON_ID) REFERENCES PERSON (PERSON_ID),
  ADDR_ID INTEGER(4), FOREIGN KEY (ADDR_ID) REFERENCES ADDRESS (ADDR_ID),
  ADDRESS_TYPE_CODE CHARACTER(1), FOREIGN KEY (ADDRESS_TYPE_CODE) REFERENCES REFM_ADDRESS_TYPE (ADDRESS_TYPE_CODE),
  ADDRESS_SOURCE_CODE CHARACTER(3), FOREIGN KEY (ADDRESS_SOURCE_CODE) REFERENCES REFM_ADDRESS_SOURCE (ADDRESS_SOURCE_CODE),
  ADDRESS_STATUS_CODE CHARACTER(1), FOREIGN KEY (ADDRESS_STATUS_CODE) REFERENCES REFM_VERIFY_STATUS (VERIFY_STATUS_CODE),
  ADDRESS_VERIFY_SOURCE_CODE CHARACTER(1), FOREIGN KEY (ADDRESS_VERIFY_SOURCE_CODE) REFERENCES REFM_ADDRESS_VERIFY_SOURCE (ADDRESS_VERIFY_SOURCE_CODE),
  ADDRESS_STATUS_DATE DATE(4),
  ADDRESS_RECEIVED_DATE DATE(4),
  PRIMARY_IND BOOLEAN(1),
  ADDRESS_VERIFIED_DATE DATE(4),
  BEGIN_DATE DATE(4),
  END_DATE DATE(4),
  COMMENT_NOTE VARCHAR(4000),
  TRANSACTION_EVENT_SEQ_NUMB BIGINT(8), FOREIGN KEY (TRANSACTION_EVENT_SEQ_NUMB) REFERENCES GLOBAL_TRANS_EVENT_SEQUENCE (TRANSACTION_EVENT_SEQ_NUMB),
  UPDATED_BY_ID INTEGER(4), FOREIGN KEY (UPDATED_BY_ID) REFERENCES USER_DETAIL (USER_ID),
  BEGIN_VALIDITY_DTTM TIMESTAMP(13,12),
  END_VALIDITY_DTTM TIMESTAMP(13,12),
  SYS_TRANS_START_DTTM TIMESTAMP(13,12),
)
