CREATE TABLE CITY (
  CITY_ID INTEGER(4) PRIMARY KEY,
  STATE_CODE CHARACTER(2), FOREIGN KEY (STATE_CODE) REFERENCES ADDRESS_STATE (STATE_CODE),
  COUNTRY_CODE CHARACTER(2), FOREIGN KEY (COUNTRY_CODE) REFERENCES REFM_COUNTRY (COUNTRY_CODE),
  CITY_NAME VARCHAR(50),
  TRANSACTION_EVENT_SEQ_NUMB BIGINT(8), FOREIGN KEY (TRANSACTION_EVENT_SEQ_NUMB) REFERENCES GLOBAL_TRANS_EVENT_SEQUENCE (TRANSACTION_EVENT_SEQ_NUMB),
  UPDATED_BY_ID INTEGER(4), FOREIGN KEY (UPDATED_BY_ID) REFERENCES USER_DETAIL (USER_ID),
  BEGIN_VALIDITY_DTTM TIMESTAMP(13,12),
  END_VALIDITY_DTTM TIMESTAMP(13,12),
  SYS_TRANS_START_DTTM TIMESTAMP(13,12),
)
