CREATE TABLE PERSON_PHONE_DETAIL (
  PERSON_PHONE_ID INTEGER(4) PRIMARY KEY,
  PERSON_ID BIGINT(8), FOREIGN KEY (PERSON_ID) REFERENCES PERSON (PERSON_ID),
  PHONE_TYPE_CODE CHARACTER(1), FOREIGN KEY (PHONE_TYPE_CODE) REFERENCES REFM_PHONE_TYPE (PHONE_TYPE_CODE),
  DEVICE_TYPE_CODE CHARACTER(1), FOREIGN KEY (DEVICE_TYPE_CODE) REFERENCES REFM_DEVICE_TYPE (DEVICE_TYPE_CODE),
  TEXT_MESSAGE_ALLOWED_CODE CHARACTER(1), FOREIGN KEY (TEXT_MESSAGE_ALLOWED_CODE) REFERENCES REFM_YES_NO (YES_NO_CODE),
  PHONE_NUMB VARCHAR(15),
  PHONE_EXTN_NUMB CHARACTER(6),
  PERSON_COUNTRY_STD_NUMB CHARACTER(5),
  TRANSACTION_EVENT_SEQ_NUMB BIGINT(8), FOREIGN KEY (TRANSACTION_EVENT_SEQ_NUMB) REFERENCES GLOBAL_TRANS_EVENT_SEQUENCE (TRANSACTION_EVENT_SEQ_NUMB),
  UPDATED_BY_ID INTEGER(4), FOREIGN KEY (UPDATED_BY_ID) REFERENCES USER_DETAIL (USER_ID),
  BEGIN_VALIDITY_DTTM TIMESTAMP(13,12),
  END_VALIDITY_DTTM TIMESTAMP(13,12),
  SYS_TRANS_START_DTTM TIMESTAMP(13,12),
)
